  #include string.h
  #include conio.h
  #include dos.h
  void main() {
  char binary[33];
  clrscr();
  cout << "Enter a binary number: ";
  cin >> binary;
  int decimal = 0;
  int length = strlen(binary);
  for (int i = 0; i < length; i++) {
  if (binary[i] == '1') {
  decimal += (1 << (length - 1 - i));
  } else if (binary[i] != '0') {
  cout << "Invalid binary number!" << endl;
  return 1;
  }
  }
  cout << "The decimal value is: " << decimal << endl;
  delay(5000);
  }

13

  #include iostream.h
  #include cstdlib.h
  #include conio.h
  #include dos.h
  void main() {
  int decimal;
  clrscr();
  cout << "Enter a decimal number: ";
  cin >> decimal;
  if (decimal < 0) {
  cout << "Please enter a non-negative integer." << endl;
  exit(1);
  }
  int binary[32];
  int index = 0;
  if (decimal == 0) {
  cout << "The binary value is: 0" << endl;
  return 0;
  }
  while (decimal > 0) {
  binary[index] = decimal % 2; 
  decimal = decimal / 2;       
  index++;
  }
  cout << "The binary value is: ";
  for (int i = index - 1; i >= 0; i--) {
  cout << binary[i];
  }
  cout << endl;
  delay(5000);
  }

13

Output:

1101
