void sortString(char str[]) {
    int i, j;
    char temp;
    int len = strlen(str);
    for (i = 0; i < len - 1; i++) {
        for (j = i + 1; j < len; j++) {
            if (str[i] > str[j]) {
                temp = str[i];
                str[i] = str[j];
                str[j] = temp;
            }
        }
    }
}
void toLowerCase(char str[]) {
    int i;

        str[i] = tolower(str[i]);

}

void permute(char str[], int l, int r) {
    if (l == r) {
        cout << str << endl;
    } else {
        for (int i = l; i <= r; i++) {
            swap(str[l], str[i]);  
            permute(str, l + 1, r);  
            swap(str[l], str[i]); 
        }
    }
}

                strcpy(smallest, word);

    }

    int n = strlen(str);

            char currentSubstr[100];
            strncpy(currentSubstr, &str[i], len);

            for (int j = i + 1; j <= n - len; j++) {

                    }
                    break;  
                }

        << longestRepeating << endl;

#include iostream.h

#include fstream.h
#include string.h
#include stdlib.h

#define MAX_WORDS 1000
void findMostRepeatedWord(char *filename) {
    ifstream file(filename);
    if (!file) {
        cout << "Error opening file!" << endl;
        return;
    }
    char word[100], mostRepeatedWord[100];
    int maxCount = 0;  
    int wordCount[MAX_WORDS] = {0};  
    char words[MAX_WORDS][100];
    int totalWords = 0;
    while (file >> word) {
        for (int i = 0; word[i]; i++) {
            word[i] = tolower(word[i]);
        }
        bool found = false;
        for (int i = 0; i < totalWords; i++) {
            if (strcmp(words[i], word) == 0) {
                wordCount[i]++;
                found = true;
                break;
            }
        }
        if (!found) {
            strcpy(words[totalWords], word);
            wordCount[totalWords] = 1;
            totalWords++;
        }
    }
    for (int i = 0; i < totalWords; i++) {
        if (wordCount[i] > maxCount) {
            maxCount = wordCount[i];
            strcpy(mostRepeatedWord, words[i]);
        }
    }
    if (maxCount > 0) {
        cout << "The most repeated word is: " 
        << mostRepeatedWord << endl;
        cout << "It occurs " << maxCount 
        << " times." << endl;
    } else {
        cout<< "No words found in the file."<< endl;
    }

    file.close();
}
void main() {
    clrscr();
    char filename[100];
  cout<<"Enter the filename (with extension, e.g., 'sample.txt'): ";
    cin >> filename;
    findMostRepeatedWord(filename);

int isPalindrome(char word[]) {
    int len = strlen(word);
    for (int i = 0; i < len / 2; i++) {
        if (word[i] != word[len - 1 - i])

    return 1;
}

void sortNames(char names[][50], int n) {
    char temp[50];
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            if (strcmp(names[i], names[j]) > 0) {
                strcpy(temp, names[i]);
                strcpy(names[i], names[j]);
                strcpy(names[j], temp);
            }
        }
    }
}
