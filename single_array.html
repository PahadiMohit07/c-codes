    cout << "Enter number of elements in the array: ";
    cin >> n;
    cout << "Enter elements:\n";
    for(i = 0; i < n; i++)
    {
        cout << "arr[" << i << "] = ";
        cin >> arr[i];
        freq[i] = -1;
    }
    for(i = 0; i < n; i++)
    {
        count = 1;
        for(j = i + 1; j < n; j++)
        {
            if(arr[i] == arr[j])
            {
                count++;
                freq[j] = 0; 
            }
        }
        if(freq[i] != 0)
        {
            freq[i] = count;
        }
    }
    cout << "\nFrequency of each element:\n";
    for(i = 0; i < n; i++)
    {
        if(freq[i] != 0)
        { 
       cout << arr[i] << " occurs " << freq[i] << " time(s)\n";
        }
    }
  delay(5000);
}
