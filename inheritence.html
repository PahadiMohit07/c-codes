  #include iostream.h
  #include conio.h
  #include dos.h
  class BankAccount {
  protected:
  float balance;              
  public:
  BankAccount(float initialBalance) {
  balance = initialBalance;
  }
  void deposit(float amount) {
  if (amount > 0) {
  balance += amount;
  cout << "Deposited: " << amount << endl;
  } else {
  cout << "Invalid deposit amount." << endl;
  }
  }
  virtual void withdraw(float amount) {
  if (amount > 0 && amount <= balance) {
  balance -= amount;
  cout << "Withdrawn: " << amount << endl;
  } else {
  cout << "Invalid withdrawal amount or insufficient 
  balance."<< endl;
  }
  }
  void displayBalance() {
  cout << "Current Balance: " << balance << endl;
  }
  };
  class SavingsAccount : public BankAccount {
  public:
  SavingsAccount(float initialBalance):
  BankAccount(initialBalance){}
  void withdraw(float amount) override {
  if (amount > 0 && (balance - amount) >= 100) {
  balance -= amount;
  cout << "Withdrawn: " << amount << endl;
  } else if ((balance - amount) < 100) {
  cout << "Withdrawal denied. Balance cannot 
  fall below 100."<< endl;
  } else {
  cout << "Invalid withdrawal amount." << endl;
  }
  }
  };
  void main() {
  clrscr();
  SavingsAccount myAccount(500);
  myAccount.displayBalance();
  myAccount.deposit(200);
  myAccount.withdraw(300); 
  myAccount.withdraw(400);
  myAccount.displayBalance();
  delay(5000);
  }

Output:

Current Balance: 500, Deposited: 200, Withdrawn: 300, Withdrawal denied. Balance cannot fall below 100., Current Balance: 400
