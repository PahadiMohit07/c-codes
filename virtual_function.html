  #include iostream.h
  #include conio.h
  #include math.h
  #include dos.h
  class Shape {
  public:
  virtual float calculateArea() = 0;          
  virtual float calculatePerimeter() = 0;     
  };
  class Circle : public Shape {
  private:
  float radius;
  public:
  Circle(float r) : radius(r) {}
  float calculateArea() {
  return 3.14 * radius * radius;  
  }
  float calculatePerimeter() {
  return 2 * 3.14 * radius;  
  }
  };
  class Triangle : public Shape {
  private:
  float a, b, c;  
  public:
  Triangle(float side1, float side2, float side3) : 
  a(side1), b(side2), c(side3) {}
  float calculateArea() {
  float s = (a + b + c) / 2;  
  return sqrt(s * (s - a) * (s - b) * (s - c));  
  }
  float calculatePerimeter() {
  return a + b + c;  
  }
  };
  void main() {
  clrscr();
  Shape* shape1 = new Circle(5);
  Shape* shape2 = new Triangle(3, 4, 5);
  cout<<"Circle Area: "<< shape1->calculateArea()<< endl;
  cout<<"Circle Perimeter: "<< shape1->calculatePerimeter()<< 
  endl;
  cout<< "Triangle Area: "<< shape2->calculateArea()<< endl;
  cout<<"Triangle Perimeter: "<< shape2->calculatePerimeter()<< 
  endl;
  delay(5000);
  }

Input:

Circle(5), Triangle(3, 4, 5)

Output:

Circle Area: 78.5
Circle Perimeter: 31.4

  cout<< "Sphere Surface Area: " << surfaceArea << endl;

  }
